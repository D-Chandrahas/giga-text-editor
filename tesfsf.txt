#include <ncurses.h>
#include <string>

int main(){
	initscr(); /* Start curses mode */
	noecho(); /* Don't echo characters */
	raw(); /* Line buffering disabled */
	keypad(stdscr, TRUE); /* Use keypad */
	int xmax, ymax;
	getmaxyx(stdscr, ymax, xmax); /* Get screen size */
	printw("%d %d\n%d %d\n",LINES,COLS,ymax,xmax); /* Print size of screen */
	addch('C' | A_REVERSE);
	int ch;
	addch('\n');
	std::string inp =  "frsd";
	inp.clear();
	inp += char(3);
	//char cancel = char(3);
	char cancel_str[2] = {char(3),'\0'};
	if(inp == cancel_str){
		printw("%s\n","true");
	}
	// if(inp == std::string(1,char(3))){
	// 	printw("%s\n","true");
	// }
	addch('\n');
	for(int i = 0; i < COLS; i++){
				addch(' ');
			}
			addch('\n');
			refresh();

	while((ch = getch())!='`'){
		printw("%d", ch);
		refresh();
	} /* Wait for user input */
	endwin(); /* End curses mode */
	return 0;
}
